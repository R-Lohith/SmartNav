################################################################################
# ① PYTHON DEPENDENCIES  (client/TamilWards/)                                 #
################################################################################

# ── Web Framework ──────────────────────────────────────────────────────────── #
Flask                   # Lightweight WSGI web application framework
Flask-CORS              # Cross-Origin Resource Sharing support for Flask

# ── Data Processing & Analysis ─────────────────────────────────────────────── #
pandas                  # DataFrame manipulation, Excel reading/writing
openpyxl                # Engine for reading/writing .xlsx files with pandas
xlsxwriter              # Engine used with pd.ExcelWriter for styled Excel output

# ── Geospatial Analysis ────────────────────────────────────────────────────── #
geopandas               # Geospatial data in Pandas DataFrames (GeoJSON, shapefiles)
shapely                 # Geometric operations (Point, LineString, Polygon, intersects)
pyproj                  # Cartographic projections (dependency of geopandas)
fiona                   # Read/write geospatial data formats (dependency of geopandas)

# ── Interactive Map Generation ─────────────────────────────────────────────── #
folium                  # Interactive Leaflet.js maps generated from Python

# ── Web Scraping – Requests ────────────────────────────────────────────────── #
requests                # HTTP library for fetching web pages and APIs
beautifulsoup4          # HTML/XML parsing for web scraping (The Hindu, TNIE, etc.)
lxml                    # Fast XML/HTML parser used by BeautifulSoup (xml parser)

# ── Web Scraping – Browser Automation ─────────────────────────────────────── #
selenium                # Browser automation for JavaScript-heavy news sites
webdriver-manager       # Automatically manages ChromeDriver binary versions

# ── Concurrency ────────────────────────────────────────────────────────────── #
# concurrent.futures is part of the Python standard library (no pip install)
